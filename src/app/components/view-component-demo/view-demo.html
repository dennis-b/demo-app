Angular View simple example.

This is regular angular directive that using: bindToController and controllerAs syntax
The class will be the controller and exposed as vm object on the scope.
Create View directive. by default using Class name as Directive name : 'ng-view-demo'
Or you can specify selector in View options 
import   './style/view-demo.css'
import template from './view-demo.html';
import {View} from 'annotations/directive-decorator';

@View({
    selector: 'ng-view-demo',
    template: template
})
//
class NgViewDemo {
    constructor($window) {
        this.$window = $window;//service injected
        this.items = ['One', 'Two', 'Three'];//using the object for  directive state
    }
}

export default NgViewDemo
